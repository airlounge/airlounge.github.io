import numpy as np
import matplotlib.pyplot as plt

def decompose_wind_vector(wind, front_vector):
    """
    ê·¸ëŒ-ìŠˆë¯¸íŠ¸ ì›ë¦¬ë¥¼ ì´ìš©í•˜ì—¬ ë°”ëŒ ë²¡í„°ë¥¼ ì „ì„ ì— í‰í–‰í•œ ì„±ë¶„ê³¼ ìˆ˜ì§ì¸ ì„±ë¶„ìœ¼ë¡œ ë¶„ë¦¬
    """
    # 1. ì „ì„  ë²¡í„° ì •ê·œí™” (ë°©í–¥ ë²¡í„° ë§Œë“¤ê¸°)
    f_mag = np.linalg.norm(front_vector)
    if f_mag == 0: return wind, wind
    f_unit = front_vector / f_mag
    
    # 2. ì „ì„  í‰í–‰ ì„±ë¶„ (Projection) -> ìˆ˜í•™: ì •ì‚¬ì˜
    parallel_mag = np.dot(wind, f_unit)
    v_parallel = parallel_mag * f_unit
    
    # 3. ì „ì„  ìˆ˜ì§ ì„±ë¶„ (Orthogonal) -> ìˆ˜í•™: ì§êµí™”
    v_perp = wind - v_parallel
    
    return v_parallel, v_perp

# --- ë°ì´í„° ìƒì„± ---
x, y = np.meshgrid(np.linspace(-5, 5, 10), np.linspace(-5, 5, 10))
front_vec = np.array([1, -1]) # ë¶ì„œ-ë‚¨ë™ ë°©í–¥ ì „ì„ 

# ë¶í’(North Wind) ìƒì„±
u = np.zeros_like(x)
v = np.full_like(y, -1.0) 

# --- ë²¡í„° ë¶„í•´ ìˆ˜í–‰ ---
u_par, v_par = np.zeros_like(u), np.zeros_like(v)
u_perp, v_perp = np.zeros_like(u), np.zeros_like(v)

for i in range(u.shape[0]):
    for j in range(u.shape[1]):
        current_wind = np.array([u[i, j], v[i, j]])
        vp, vn = decompose_wind_vector(current_wind, front_vec)
        u_par[i, j], v_par[i, j] = vp[0], vp[1]
        u_perp[i, j], v_perp[i, j] = vn[0], vn[1]

# --- ì‹œê°í™” ---
fig, axes = plt.subplots(1, 3, figsize=(18, 6))
x_line = np.linspace(-4, 4, 100)
y_line = -x_line 

titles = ["(A) Observed Wind (Total)", 
          "(B) Parallel Component (Shear)", 
          "(C) Perpendicular Component (Lifting)"]
data = [(u, v), (u_par, v_par), (u_perp, v_perp)]
colors = ['black', 'blue', 'red']

for ax, title, (du, dv), color in zip(axes, titles, data, colors):
    ax.plot(x_line, y_line, 'b-', linewidth=3, alpha=0.5, label='Front')
    ax.quiver(x, y, du, dv, color=color, scale=15, width=0.005)
    ax.set_title(title, fontsize=12, fontweight='bold')
    ax.set_xlim(-5, 5); ax.set_ylim(-5, 5)
    ax.grid(True, linestyle=':'); ax.set_aspect('equal')
    ax.legend()

plt.suptitle(f"Gram-Schmidt Application: Frontal Wind Decomposition", fontsize=16)
plt.tight_layout()
plt.show()
